import { MetadataRoute } from "next";

export default function robots(): MetadataRoute.Robots {
  return {
    rules: [
      {
        userAgent: "*",
        allow: "/",
        disallow: [], // nothing blocked — all pages are public
      },
    ],
    sitemap: "https://www.knowfut.com/sitemap.xml",
    host: "https://www.knowfut.com",
  };
}
